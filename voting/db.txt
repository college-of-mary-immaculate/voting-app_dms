live results:
	SELECT 
    		p.position_name,
    		c.fullname,
    		COUNT(v.vote_id) AS total_votes
	FROM candidates c
	LEFT JOIN votes v 
    		ON c.candidate_id = v.candidate_id
	JOIN positions p 
    		ON c.position_id = p.position_id
	GROUP BY c.candidate_id
	ORDER BY p.position_id, total_votes DESC;
